# shapes used in this model
instance sphere models/sphere.obj
instance box models/box.obj

#material
material head-material 
ambient 1 1 0
end-material

material body-material 
ambient 0 0 1
end-material

material eye-material
ambient 0 1 0
end-material

group drone drone

#drone body
scale s-body s-body 100 25 50
add-child s-body drone
leaf body body instanceof box
add-child body s-body 
assign-material body body-material 

#drone head

add-child t-head drone
scale s-head s-head 10 15 40
add-child s-head t-head

add-child head s-head
